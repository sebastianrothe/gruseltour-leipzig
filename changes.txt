/**
 * Slideshow stylen
 */
function style_slideshow() {
    ?>
<script type="text/javascript">
    function captionate($images) {
        $images.each(function () {
            var $caption = jQuery(this).attr('alt');
            jQuery(this).wrap('<figure></figure>');
            jQuery(this).after('<figcaption>' + $caption + '</figcaption>');
        });
    }

    function slideSwitch($images) {
        var $active = $images.filter('img.active');
        if ($active.length === 0) $active = $images.last();

        //var $next = $images.filter('img.active ~ img').first();
        var $next = $images.eq($images.index($active) + 1);
        if ($next.length === 0) $next = $images.first();

        var $activeCaption = jQuery($active).siblings().first();
        var $nextCaption = jQuery($next).siblings().first();

        $active.add($activeCaption).addClass('last-active');

        $nextCaption.css({
            opacity: 0.0
        })
            .addClass('active')
            .animate({
            opacity: 0.5
        }, {
            duration: 2000,
            easing: "swing",
            start: function () {
                $activeCaption.animate({
                    opacity: 0.0
                }, {
                    duration: 1000,
                    easing: "swing"
                });
            },
            complete: function () {
                $activeCaption.removeClass('active last-active');
            }
        });

        $next.css({
            opacity: 0.0
        })
            .addClass('active')
            .animate({
            opacity: 1.0
        }, {
            duration: 2000,
            easing: "swing",
            start: function () {
                $active.animate({
                    opacity: 0.0
                }, {
                    duration: 2000,
                    easing: "swing"
                });
            },
            complete: function () {
                $active.removeClass('active last-active');
            }
        });

        setTimeout(function () {
            slideSwitch($images);
        }, 5000);
    }

    function activateSlideshow($images) {
        captionate($images);

        setTimeout(function () {
            slideSwitch($images);
        }, 5000);
    }

    jQuery(document).ready(function () {
        activateSlideshow(jQuery('#slideshow img'));
    });
</script>
    <?php
}
add_action( 'wp_footer', 'style_slideshow');
